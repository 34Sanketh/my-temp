# --- Prediction Function ---

def predict_questions(user_query, custom_instruction=None):
    """
    This is the engine of your system. It searches the RAG system
    and then uses the gemini-2.5-flash model to generate a prediction.
    """
    # 1. Search your RAG database for relevant information
    retriever = vectorstore.as_retriever(search_kwargs={"k": 5})
    relevant_docs = retriever.get_relevant_documents(user_query)
    context = "\\n\\n---\\n\\n".join([doc.page_content for doc in relevant_docs])

    # 2. Build a powerful prompt for the LLM
    base_instruction = "Act as an expert exam analyst. Based on the provided context, analyze the style, topics, and difficulty of the questions in the past papers."
    final_instruction = custom_instruction if custom_instruction else base_instruction

    prompt = f"""{final_instruction}

    Context (from your textbooks and past papers):
    {context}

    Task for the LLM: {user_query}

    Generate a detailed and accurate response based only on the provided context.
    """

    # 3. Call the gemini-2.5-flash model to generate the answer
    model = genai.GenerativeModel('gemini-2.5-flash')
    response = model.generate_content(prompt)

    return response.text

def check_question_online(question_text, api_key=CSE_KEY, cse_id=CSE_ID):
    """Returns (status, url) where status is 'Likely Original' or 'Found Online'"""
    try:
        service = build("customsearch", "v1", developerKey=api_key)
        res = service.cse().list(q=question_text, cx=cse_id, num=1).execute()
        if 'items' in res:
            return "Found Online", res['items'][0]['link']
        else:
            return "Likely Original", None
    except Exception as e:
        print(f"Search error: {e}")
        return "Check Failed", None

def check_past_paper(subject, filename):
    # Load PDF and extract questions (simplified: extract all text, then split by question numbers)
    pdf_path = f"{BASE_DIR}/past_papers/{subject}/{filename}"
    loader = PyPDFLoader(pdf_path)
    pages = loader.load()
    full_text = "\n".join([p.page_content for p in pages])
    
    # Naive split: look for "Q1.", "Q2." etc.
    questions = re.split(r'\nQ\d+\.', full_text)
    questions = [q.strip() for q in questions if len(q) > 20][1:]  # skip first empty
    
    results = []
    for i, q in enumerate(questions[:10]):  # limit to 10 to save quota
        status, url = check_question_online(q[:200])  # first 200 chars
        results.append({"Q": i+1, "question_preview": q[:100], "status": status, "url": url})
        time.sleep(1)  # avoid rate limits
    
    return pd.DataFrame(results)

def build_subject_index(subject):
    """Create or update ChromaDB index for a subject"""
    print(f"🔨 Building index for {subject}...")
    
    # Load textbooks
    textbook_dir = f"{BASE_DIR}/textbooks/{subject}"
    all_docs = []
    for pdf in os.listdir(textbook_dir):
        if pdf.endswith(".pdf"):
            loader = PyPDFLoader(f"{textbook_dir}/{pdf}")
            all_docs.extend(loader.load())
    
    # Load past papers (optional, can include for context)
    papers_dir = f"{BASE_DIR}/past_papers/{subject}"
    for pdf in os.listdir(papers_dir):
        if pdf.endswith(".pdf"):
            loader = PyPDFLoader(f"{papers_dir}/{pdf}")
            all_docs.extend(loader.load())
    
    if not all_docs:
        print("No documents found for this subject.")
        return None
    
    # Split into chunks
    splitter = RecursiveCharacterTextSplitter(chunk_size=1500, chunk_overlap=200)
    chunks = splitter.split_documents(all_docs)
    
    # Embed and store
    embeddings = HuggingFaceEmbeddings(model_name="sentence-transformers/all-MiniLM-L6-v2")
    vectorstore = Chroma.from_documents(chunks, embeddings, persist_directory=f"{BASE_DIR}/vectorstore/{subject}")
    print(f"✅ Index for {subject} built with {len(chunks)} chunks.")
    return vectorstore

def extract_pattern(subject, source, teacher_name=None):
    """
    source: 'College Teacher' or 'Government'
    teacher_name: required if source is College Teacher
    """
    pattern_file = f"{BASE_DIR}/patterns/{subject}/{source}_{teacher_name if teacher_name else 'govt'}.json"
    
    # Load metadata for past papers of this source
    meta_path = f"{BASE_DIR}/past_papers/{subject}/metadata.json"
    if not os.path.exists(meta_path):
        print(f"No past papers found for {subject} {source}")
        return None
    
    with open(meta_path, "r") as f:
        all_meta = json.load(f)
    
    relevant_papers = [m for m in all_meta if m['source'] == source]
    if source == "College Teacher":
        relevant_papers = [m for m in relevant_papers if m.get('teacher_name') == teacher_name]
    
    if not relevant_papers:
        print(f"No papers found for {subject} {source} {teacher_name or ''}")
        return None
    
    # Extract text from those papers (simplified: just concatenate)
    full_text = ""
    for paper in relevant_papers:
        pdf_path = f"{BASE_DIR}/past_papers/{subject}/{paper['filename']}"
        loader = PyPDFLoader(pdf_path)
        pages = loader.load()
        full_text += "\n".join([p.page_content for p in pages])
    
    # Use Gemini to analyze pattern
    prompt = f"""
    You are an exam pattern analyzer. Analyze these past exam papers for {subject} (source: {source}).
    Extract:
    1. Top 10 most frequent topics (list as bullet points)
    2. Typical question types (e.g., MCQ, short answer, essay)
    3. Average number of questions
    4. Any repeating questions or patterns (e.g., topic X appears every 2 exams)
    Return as JSON format.
    
    Text from past papers:
    {full_text[:15000]}  # limit to 15k tokens
    """
    model = genai.GenerativeModel('gemini-2.5-flash')
    response = model.generate_content(prompt)
    
    pattern = {
        "subject": subject,
        "source": source,
        "teacher": teacher_name,
        "analysis_date": str(datetime.now().date()),
        "raw_analysis": response.text
    }
    
    with open(pattern_file, "w") as f:
        json.dump(pattern, f, indent=2)
    print(f"✅ Pattern saved to {pattern_file}")
    return pattern

def predict_next_exam():
    # Find next exam from schedule
    df = pd.read_csv(schedule_path)
    df['date'] = pd.to_datetime(df['date'])
    today = datetime.now().date()
    next_exam = df[df['date'].dt.date >= today].sort_values('date').iloc[0]
    
    subject = next_exam['subject']
    exam_type = next_exam['exam_type']
    source = next_exam['source']
    exam_date = next_exam['date'].date()
    
    print(f"\n🔮 Predicting for {subject} on {exam_date} ({exam_type} - {source})")
    
    # Load pattern
    if source == "College Teacher":
        # Need teacher name - for now assume you will input or we extract from metadata
        # For simplicity, we'll ask user to enter teacher name
        teacher_name = input(f"Enter teacher name for {subject} (e.g., Dr. Sharma): ")
        pattern_file = f"{BASE_DIR}/patterns/{subject}/College Teacher_{teacher_name}.json"
    else:
        pattern_file = f"{BASE_DIR}/patterns/{subject}/Government_govt.json"
    
    pattern = {}
    if os.path.exists(pattern_file):
        with open(pattern_file, "r") as f:
            pattern = json.load(f)
    else:
        print("⚠️ No pattern found. Running pattern extraction first...")
        if source == "College Teacher":
            teacher_name = input(f"Enter teacher name for {subject}: ")
            pattern = extract_pattern(subject, source, teacher_name)
        else:
            pattern = extract_pattern(subject, source)
    
    # Load textbook index for context
    vectorstore = Chroma(persist_directory=f"{BASE_DIR}/vectorstore/{subject}", embedding_function=HuggingFaceEmbeddings(model_name="sentence-transformers/all-MiniLM-L6-v2"))
    # Retrieve important topics (using a generic query)
    retriever = vectorstore.as_retriever(search_kwargs={"k": 10})
    important_topics_docs = retriever.get_relevant_documents("most important topics for exam")
    textbook_context = "\n".join([d.page_content[:500] for d in important_topics_docs])
    
    # Build prediction prompt
    if source == "College Teacher":
        prompt = f"""
        You are an expert exam predictor. Predict the next exam paper for {subject}.
        Exam type: {exam_type}, set by a college teacher named {teacher_name}.
        
        Teacher's pattern (from past papers):
        {pattern.get('raw_analysis', 'No pattern available')}
        
        Important topics from textbook:
        {textbook_context[:10000]}
        
        Generate a predicted question paper with exactly the same style, number of questions, and difficulty as the teacher's past papers.
        Return a list of predicted questions (as numbered list). For each question, give a confidence score (0-100) and a brief reason.
        Also list any "surprise" topics (important but never asked before).
        """
    else:  # Government
        prompt = f"""
        You are an expert exam predictor. Predict the next {exam_type} exam for {subject}.
        This exam is set by government experts.
        
        Pattern from past government exams:
        {pattern.get('raw_analysis', 'No pattern available')}
        
        Important topics from textbook (based on syllabus weightage):
        {textbook_context[:10000]}
        
        Generate a predicted question paper that matches the standard format of government exams (balanced topics, moderate difficulty).
        Return a list of predicted questions. For each, give a confidence score.
        Also list the topic weightage distribution.
        """
    
    model = genai.GenerativeModel('gemini-2.5-flash')
    response = model.generate_content(prompt)
    
    # Save prediction
    pred_file = f"{BASE_DIR}/predictions/{subject}_{exam_date}_{exam_type}.txt"
    with open(pred_file, "w") as f:
        f.write(response.text)
    print(f"✅ Prediction saved to {pred_file}")
    
    # Also generate PDF
    generate_prediction_pdf(subject, exam_date, exam_type, response.text)
    
    return response.text

def generate_prediction_pdf(subject, exam_date, exam_type, content):
    pdf = FPDF()
    pdf.add_page()
    pdf.set_font("Arial", size=12)

    pdf.ln(10)
    pdf.multi_cell(0, 10, content)
    pdf_file = f"{BASE_DIR}/predictions/{subject}_{exam_date}_{exam_type}.pdf"
    pdf.output(pdf_file)
    print(f"📄 PDF saved to {pdf_file}")

def check_question_online(question_text):
    """Check if a question exists online via Google Custom Search."""
    try:
        service = build("customsearch", "v1", developerKey=CSE_KEY)
        res = service.cse().list(q=question_text[:200], cx=CSE_ID, num=1).execute()
        if 'items' in res:
            return "Found Online", res['items'][0]['link']
        else:
            return "Likely Original", None
    except Exception:
        return "Check Failed", None

    if not pdf_files:
        st.warning("No PDF files found.")
        return
    selected_pdf = st.selectbox("Select Past Paper", pdf_files)
    if st.button("Check for Copied Questions"):
        with st.spinner("Processing..."):
            loader = PyPDFLoader(f"{past_papers_dir}/{selected_pdf}")
            pages = loader.load()
            full_text = "\n".join([p.page_content for p in pages])
            questions = re.split(r'\nQ\d+\.', full_text)
            questions = [q.strip() for q in questions if len(q) > 20][1:]
            results = []
            for i, q in enumerate(questions[:10]):
                status, url = check_question_online(q[:200])
                results.append({"Q": i+1, "Preview": q[:100], "Status": status, "URL": url if url else ""})
                time.sleep(1)
            df = pd.DataFrame(results)
            st.dataframe(df)

import requests

def check_question_online(question_text):
    # 1. Use a public SearXNG instance to search for the question
    searxng_instance = "https://searx.party"  # Free, public instance
    search_url = f"{searxng_instance}/search?q={question_text}&format=json"
    
    try:
        # 2. Get the search results
        response = requests.get(search_url, headers={"User-Agent": "Mozilla/5.0"})
        results = response.json()
        
        if results.get('results'):
            top_result_url = results['results'][0]['url']
            
            # 3. Use Jina.ai Reader to get the content of the top search result
            reader_url = f"https://r.jina.ai/{top_result_url}"
            content_response = requests.get(reader_url)
            
            # 4. Analyze the content to see if it matches the question
            # This is a placeholder check
            if question_text.lower() in content_response.text.lower():
                return "Found Online", top_result_url
            else:
                return "Likely Original", None
    except Exception as e:

    return "Likely Original", None

import streamlit as st
import json
from google.oauth2 import service_account
from googleapiclient.discovery import build
from googleapiclient.http import MediaIoBaseDownload
import io

def authenticate_drive():
    """Authenticate with Google Drive using the service account from secrets."""
    service_account_info = json.loads(st.secrets["google_service_account_key"])
    credentials = service_account.Credentials.from_service_account_info(
        service_account_info,
        scopes=["https://www.googleapis.com/auth/drive"]
    )
    return build("drive", "v3", credentials=credentials)

def list_files_in_folder(folder_id):
    """List all files in a specific Google Drive folder."""
    service = authenticate_drive()
    query = f"'{folder_id}' in parents and trashed=false"
    results = service.files().list(q=query, fields="files(id, name)").execute()
    return results.get("files", [])

def download_file(file_id):
    """Download a file from Google Drive as bytes."""
    service = authenticate_drive()
    request = service.files().get_media(fileId=file_id)
    file = io.BytesIO()
    downloader = MediaIoBaseDownload(file, request)
    done = False
    while not done:
        status, done = downloader.next_chunk()
    file.seek(0)
    return file
